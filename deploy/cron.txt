# Cron job untuk shared hosting (cPanel)
#
# Buka cPanel → Cron Jobs → tambahkan entri berikut (setiap menit):
#
# * * * * * /usr/local/bin/php /home/USERNAME/kkn_app/artisan schedule:run >> /dev/null 2>&1
#
# Ganti USERNAME dengan username cPanel Anda.
# Path php bisa berbeda — cek di cPanel atau jalankan: which php
#
# Tugas terjadwal:
# - absensi:rotate-token setiap hari pukul 00:05 (generate token QR absensi baru)
#
# Catatan: token juga dibuat otomatis saat halaman QR dibuka, tetapi cron
# memastikan token tersedia tepat waktu tanpa perlu buka halaman manual.
